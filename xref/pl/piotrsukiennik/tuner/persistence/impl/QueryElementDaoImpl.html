22      public GroupByFragment getOrCreateGroupBy( GroupByFragment groupByFragment ) {
23          GroupByFragment groupByFragmentPer = (GroupByFragment) s().createCriteria( GroupByFragment.class )
24           .add( Restrictions.eq( "expression", groupByFragment.getExpression() ) )
25           .add( Restrictions.eq( "position", groupByFragment.getPosition() ) ).setMaxResults( 1 ).uniqueResult();
26          if ( groupByFragmentPer == null ) {
27              return create( groupByFragment );
28          }
29          return groupByFragmentPer;
30      }
