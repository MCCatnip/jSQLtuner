14  public class KeyValueServiceMemcachedImpl implements KeyValueService {
15      private static final long MAX_EXPIRATION = TimeUnit.DAYS.toSeconds( 30 );
16  
17      private MemcachedClient memcachedClient;
18  
19      public KeyValueServiceMemcachedImpl() {
20      }
21  
22      public KeyValueServiceMemcachedImpl( MemcachedClient memcachedClient ) {
23          this.memcachedClient = memcachedClient;
24      }
25  
26  
27      @Override
28      public void put( String key, Object value ) {
29          put( key, value, TimeUnit.SECONDS, MAX_EXPIRATION );
30      }
31  
32      @Override
33      public Object get( String key ) {
34          return memcachedClient.get( key );
35      }
36  
37      @Override
38      public void put( String key, Object value, TimeUnit timeUnit, long expiration ) {
39          memcachedClient.set( key, (int) ( timeUnit.toSeconds( expiration ) ), value );
40      }
41  
42      @Override
43      public void delete( String key ) {
44          memcachedClient.delete( key );
45      }
46  
47      @Override
48      public String getIdentifier() {
49          String id = "";
50          String comma = "";
51          for ( SocketAddress socketAddress : memcachedClient.getAvailableServers() ) {
52              id += comma + socketAddress.toString();
53              comma = ",";
54          }
55          return id;
56      }
57  }
