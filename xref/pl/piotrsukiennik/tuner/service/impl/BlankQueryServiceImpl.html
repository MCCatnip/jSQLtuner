11  import java.util.Map;

13  /**
14   * @author Piotr Sukiennik

17  @Service
18  public class BlankQueryServiceImpl implements QueryService {
19  
20      private Map<String, Query> cachedQueries = new LinkedHashMap<>();
21  
22  
23      protected <T extends Query> T get( String hash ) {
24          return (T) cachedQueries.get( hash );
25      }
26  
27  
28      @Override
29      public <T extends Query> T getQuery( String database, String schema, String queryStr ) throws QueryParsingNotSupportedException {
30          String hash = getHash( database, schema, queryStr );
31          return get( hash );
32      }
33  
34      protected String getHash( String databaseStr, String schemaStr, String query ) {
35          return HashGenerators.MD5.getHash( databaseStr + "." + schemaStr + "." + query );
36      }
37  
38      @Override
39      public <T extends Query> void createQuery( T query, String database, String schema, String queryStr ) {
40          save( query, getHash( database, schema, queryStr ) );
41      }
42  
43  
44      protected <T extends Query> void save( T query, String hash ) {
45          query.setHash( hash );
46          cachedQueries.put( hash, query );
47      }
48  
49      @Override
50      public void createExecution( WriteQueryExecution writeQueryExecution ) {
51      }
52  }
