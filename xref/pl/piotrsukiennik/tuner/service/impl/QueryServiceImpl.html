1   package pl.piotrsukiennik.tuner.service.impl;

3   import org.springframework.stereotype.Service;
4   import pl.piotrsukiennik.tuner.exception.QueryParsingNotSupportedException;
5   import pl.piotrsukiennik.tuner.model.log.WriteQueryExecution;
6   import pl.piotrsukiennik.tuner.model.query.Query;
7   import pl.piotrsukiennik.tuner.persistence.Dao;
8   import pl.piotrsukiennik.tuner.persistence.QueryDao;
9   import pl.piotrsukiennik.tuner.service.QueryService;
10  import pl.piotrsukiennik.tuner.util.hash.HashGenerators;
11  
12  import java.util.LinkedHashMap;
13  import java.util.Map;
14  
15  /**
16   * @author Piotr Sukiennik
17   * @date 19.02.14
18   */
19  @Service
20  public class QueryServiceImpl implements QueryService {
21  
22      private Map<String, Query> cachedQueries = new LinkedHashMap<>();
23  
24      protected QueryDao getQueryDao() {
25          return Dao.getQuery();
26      }
27  
28      public <T extends Query> T getQuery( String hash ) {
29          T query = (T) cachedQueries.get( hash );
30          if ( query == null ) {
31              query = getQueryDao().getQuery( hash );
32              cachedQueries.put( hash, query );
33          }
34          return query;
35      }
36  
37  
38      @Override
39      public <T extends Query> T getQuery( String database, String schema, String queryStr ) throws QueryParsingNotSupportedException {
40          String hash = getHash( database, schema, queryStr );
41          return getQuery( hash );
42      }
43  
44      protected String getHash( String databaseStr, String schemaStr, String query ) {
45          return HashGenerators.MD5.getHash( databaseStr + "." + schemaStr + "." + query );
46      }
47  
48      @Override
49      public <T extends Query> void createQuery( T query, String database, String schema, String queryStr ) {
50          createQuery( query, getHash( database, schema, queryStr ) );
51      }
52  
53  
54      protected <T extends Query> void createQuery( T query, String hash ) {
55          query.setHash( hash );
56          getQueryDao().createQuery( query );
57          cachedQueries.put( hash, query );
58      }
59  
60      @Override
61      public void createExecution( WriteQueryExecution writeQueryExecution ) {
62          getQueryDao().createQueryExecution( writeQueryExecution );
63      }
64  }
