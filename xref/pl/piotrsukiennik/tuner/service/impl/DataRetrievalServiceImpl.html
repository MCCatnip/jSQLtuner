19  /**
20   * @author Piotr Sukiennik

22   */
23  @Service
24  public class DataRetrievalServiceImpl implements DataRetrievalService {
25  
26      @Override
27      public final DataRetrieval get( DataSource dataSource, ReadQuery query ) throws DataRetrievalException {
28          TimedCallable<CachedRowSet> timedCallable = getTimedCallable( dataSource, query );
29          try {
30              CachedRowSet cachedRowSet = timedCallable.call();
31              return new DataRetrieval( query, dataSource.getDataSourceIdentity(), cachedRowSet, timedCallable.getTime( TimeUnit.NANOSECONDS ), cachedRowSet.size() );
32          }
33          catch ( Exception e ) {
34              throw new DataRetrievalException( e, query, dataSource.getDataSourceIdentity() );
35          }
36      }
37  
38      protected TimedCallable<CachedRowSet> getTimedCallable( final DataSource dataSource, final ReadQuery query ) {
39          return new TimedCallableImpl<>( new Callable<CachedRowSet>() {
40              @Override
41              public CachedRowSet call() throws Exception {
42                  ResultSet resultSet = dataSource.get( query );
43                  if ( resultSet != null ) {
44                      try {
45                          return RowSet.clone( resultSet );
46                      }
47                      catch ( SQLException e ) {
48                          throw new DataRetrievalException( e, query, dataSource.getDataSourceIdentity() );
49                      }
50                  }
51                  throw new DataRetrievalException( "No data found for query", query, dataSource.getDataSourceIdentity() );
52              }
53          } );
54      }
55  }
