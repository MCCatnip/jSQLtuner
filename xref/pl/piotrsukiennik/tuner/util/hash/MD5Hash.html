10  /**
11   * Java program to generate MD5 hash or digest for String. In this example
12   * we will see 3 ways to create MD5 hash or digest using standard Java API,
13   * Spring framework and open source library, Apache commons codec utilities.
14   * Generally MD5 has are represented as Hex String so each of this function
15   * will return MD5 hash in hex format.
16   *
17   * @author Javin Paul
18   */
19  public class MD5Hash {
20      private static Log LOG = LogFactory.getLog( MD5Hash.class );
21  
22      public static String toMd5( String message ) {
23          String digest = null;
24          try {
25              MessageDigest md = MessageDigest.getInstance( "MD5" );
26              byte[] hash = md.digest( message.getBytes( "UTF-8" ) );
27              //converting byte array to Hexadecimal String
28              StringBuilder sb = new StringBuilder( 2 * hash.length );
29              for ( byte b : hash ) {
30                  sb.append( String.format( "%02x", b & 0xff ) );
31              }
32  
33              digest = sb.toString();
34  
35          }
36          catch ( UnsupportedEncodingException | NoSuchAlgorithmException ex ) {
37              if ( LOG.isErrorEnabled() ) {
38                  LOG.error( ex );
39              }
40              throw new HashGeneratorException( ex );
41          }
42          return digest;
43      }
