18  public class FromItemParser extends ParsingVisitor implements FromItemVisitor {
19  
20  
21      private Source source;
22  
23      public FromItemParser( JsqlParserQueryParsingContext parsingContext ) {
24          super( parsingContext );
25      }
26  
27  
28      @Override
29      public void visit( Table tableName ) {
30          source = parsingContext.getTableSource( tableName );
31      }
32  
33      @Override
34      public void visit( LateralSubSelect lateralSubSelect ) {
35          //TODO
36      }
37  
38      @Override
39      public void visit( ValuesList valuesList ) {
40          //TODO
41      }
42  
43      @Override
44      public void visit( SubSelect subSelect ) {
45          SubQuerySource subQuerySource = new SubQuerySource();
46          subQuerySource.setAlias( parsingContext.getTableAlias( subSelect.getAlias() ) );
47          subQuerySource.setValue( subSelect.getSelectBody().toString() );
48  
49          SelectQuery selectQuery = new SelectQuery();
50          SelectBodyParser<SelectQuery> selectBodyParser = new SelectBodyParser<SelectQuery>( parsingContext, selectQuery );
51          subSelect.getSelectBody().accept( selectBodyParser );
52          subQuerySource.setSelectQuery( selectQuery );
53          source = subQuerySource;
54      }
55  
56      @Override
57      public void visit( SubJoin subjoin ) {
58          SubJoinSource subJoinSource = new SubJoinSource();
59  
60          JoinFragment joinFragment = parsingContext.getJoin( subjoin.getJoin() );
61          FromItemParser fromItemParser = new FromItemParser( parsingContext );
62          subjoin.getLeft().accept( fromItemParser );
63  
64          subJoinSource.setJoinFragment( joinFragment );
65          subJoinSource.setAlias( parsingContext.getTableAlias( subjoin.getAlias() ) );
66          subJoinSource.setSource( fromItemParser.getSource() );
67          subJoinSource.setValue( subjoin.toString() );
68  
69          source = subJoinSource;
70      }
71  
72      public Source getSource() {
73          return source;
74      }
75  }
