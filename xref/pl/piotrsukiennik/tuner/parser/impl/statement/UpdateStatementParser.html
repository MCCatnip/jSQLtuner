23  public class UpdateStatementParser<U extends UpdateQuery> extends StatementParser<UpdateQuery> {
24      public UpdateStatementParser( JsqlParserQueryParsingContext parsingContext, Update update ) {
25          super( parsingContext, update, new UpdateQuery() );
26      }
27  
28      @Override
29      public void visit( Update update ) {
30          TableSource tableSource = parsingContext.getTableSource( update.getTable() );
31          List<Column> columnList = update.getColumns();
32          List<Expression> values = update.getExpressions();
33          Set<ColumnValue> valueEntities = new LinkedHashSet<ColumnValue>();
34  
35          for ( int i = 0; i < columnList.size(); i++ ) {
36              Column column = columnList.get( i );
37              Expression expression = values.get( i );
38              pl.piotrsukiennik.tuner.model.schema.Column column1 = parsingContext.getColumn( tableSource.getTable(), column );
39              ColumnValue columnValue = new ColumnValue();
40              columnValue.setColumn( column1 );
41              ValueEntityExpresionParser parser = new ValueEntityExpresionParser( parsingContext, columnValue );
42              expression.accept( parser );
43              valueEntities.add( columnValue );
44          }
45  
46          ExpresionParser expresionParser = new ExpresionParser( parsingContext );
47          update.getWhere().accept( expresionParser );
48          query.setWhereExpression( (OperatorExpression) expresionParser.getExpression() );
49          query.setColumnValues( valueEntities );
50          super.visit( update );
51      }

54  }
