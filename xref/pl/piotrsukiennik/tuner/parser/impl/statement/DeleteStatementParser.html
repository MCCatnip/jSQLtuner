15  public class DeleteStatementParser extends StatementParser<DeleteQuery> {
16      public DeleteStatementParser( JsqlParserQueryParsingContext parsingContext, Delete delete ) {
17          super( parsingContext, delete, new DeleteQuery() );
18      }
19  
20  
21      @Override
22      public void visit( Delete delete ) {
23          TableSource tableSource = parsingContext.getTableSource( delete.getTable() );
24          query.setTableSource( tableSource );
25          if ( delete.getWhere() != null ) {
26              ExpresionParser expresionParser = new ExpresionParser( parsingContext );
27              delete.getWhere().accept( expresionParser );
28              query.setWhereExpression( (OperatorExpression) expresionParser.getExpression() );
29          }
30          super.visit( delete );
31      }
32  
33  
34  }
