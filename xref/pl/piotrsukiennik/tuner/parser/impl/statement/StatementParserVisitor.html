22  public class StatementParserVisitor<T extends Query> extends StatementParser implements StatementVisitor {
23  
24  
25      public StatementParserVisitor( JsqlParserQueryParsingContext parsingContext, Statement statement ) {
26          super( parsingContext, statement );
27      }
28  
29      @Override
30      public void visit( Replace replace ) {
31          ReplaceStatementParser statementParser = new ReplaceStatementParser( parsingContext, replace );
32          setQuery( statementParser.getQuery() );
33      }
34  
35      @Override
36      public void visit( Truncate truncate ) {
37          TruncateTableStatementParser statementParser = new TruncateTableStatementParser( parsingContext, truncate );
38          setQuery( statementParser.getQuery() );
39      }
40  
41      @Override
42      public void visit( CreateTable createTable ) {
43          CreateTableStatementParser statementParser = new CreateTableStatementParser( parsingContext, createTable );
44          setQuery( statementParser.getQuery() );
45      }
46  
47      @Override
48      public void visit( Drop drop ) {
49          DropStatementParser statementParser = new DropStatementParser( parsingContext, drop );
50          setQuery( statementParser.getQuery() );
51      }
52  
53      @Override
54      public void visit( Select select ) {
55          SelectStatementParser statementParser = new SelectStatementParser( parsingContext, select );
56          setQuery( statementParser.getQuery() );
57      }
58  
59      @Override
60      public void visit( Delete delete ) {
61          DeleteStatementParser deleteStatementParser = new DeleteStatementParser( parsingContext, delete );
62          setQuery( deleteStatementParser.getQuery() );
63      }
64  
65      @Override
66      public void visit( Update update ) {
67          UpdateStatementParser<UpdateQuery> updateStatementParser = new UpdateStatementParser<UpdateQuery>( parsingContext, update );
68          setQuery( updateStatementParser.getQuery() );
69      }
70  
71      @Override
72      public void visit( Insert insert ) {
73          InsertStatementParser insertStatementParser = new InsertStatementParser( parsingContext, insert );
74          setQuery( insertStatementParser.getQuery() );
75      }
76  
77  }
