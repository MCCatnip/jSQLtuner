12      public static <T extends Comparable> int compare( T comparable1, T comparable2 ) {
13          if ( comparable1 == null && comparable2 == null ) {
14              return 0;
15          }
16          if ( comparable1 != null && comparable2 == null ) {
17              return 1;
18          }
19          if ( comparable1 == null ) {
20              return -1;
21          }
22          return comparable1.compareTo( comparable2 );
23      }
