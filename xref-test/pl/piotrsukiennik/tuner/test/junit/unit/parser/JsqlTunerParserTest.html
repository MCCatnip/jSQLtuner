23  @RunWith(SpringJUnit4ClassRunner.class)
24  @ContextConfiguration(locations = { "classpath*:jsqltuner-aop-context.xml" })
25  public class JsqlTunerParserTest {
26  
27      @Resource
28      private Parser parser;
29  
30      @Test
31      public void testParseSelectQuery() throws QueryParsingNotSupportedException {
32          String database = "testdb";
33          String schema = "testschema";
34          String query = "Select * from Test";
35  
36          Query queryParsed = parser.parse( database, schema, query );
37  
38          Assert.assertNotNull( queryParsed );
39          Assert.assertTrue( queryParsed instanceof SelectQuery );
40  
41          Source source = ( (SelectQuery) queryParsed ).getSources().iterator().next();
42  
43          Assert.assertNotNull( source );
44          Assert.assertTrue( source instanceof TableSource );
45  
46          Assert.assertTrue( "Test".equalsIgnoreCase( ( (TableSource) source ).getTable().getValue() ) );
47  
48          Expression projection = ( (SelectQuery) queryParsed ).getProjections().iterator().next();
49  
50          Assert.assertNotNull( projection );
51          Assert.assertTrue( projection instanceof StarProjection );
52  
53          Source projectionSource = ( (StarProjection) projection ).getSource();
54          Assert.assertSame( source, projectionSource );
55      }
